import django
from django.db import models
from django.http import JsonResponse, HttpResponse
from django.views import View
from django.contrib.auth.decorators import login_required
from django.shortcuts import render, redirect
from django.urls import path

class Student(models.Model):
  name  = models.CharField(max_length=100)
  roll  = models.IntegerField(unique=True)
  cgpa  = models.FloatField(default=0.0)
  dept  = models.CharField(max_length=60)
  year  = models.IntegerField(default=1)
  email = models.EmailField(unique=True)

  class Meta:
    ordering = ['roll']
    verbose_name = 'Student'

  def __str__(self):
    return f"{self.name} ({self.roll})"

class StudentListView(View):
  def get(self, request):
    students = Student.objects.all()
    return render(request, 'list.html',
      {'students': students})

  def post(self, request):
    data = request.POST
    Student.objects.create(
      name=data['name'],
      roll=data['roll'],
      cgpa=data['cgpa'],
      dept=data['dept'],
      year=data['year'],
      email=data['email'],
    )
    return redirect('student-list')

def update_student(request, pk):
  student = Student.objects.get(pk=pk)
  if request.method == 'POST':
    student.name = request.POST['name']
    student.cgpa = request.POST['cgpa']
    student.dept = request.POST['dept']
    student.save()
    return redirect('student-list')
  return render(request, 'edit.html',
    {'student': student})

def delete_student(request, pk):
  Student.objects.filter(pk=pk).delete()
  return redirect('student-list')

@login_required
def dashboard(request):
  total  = Student.objects.count()
  avg_cg = Student.objects.aggregate(
    avg=models.Avg('cgpa')
  )['avg']
  return render(request, 'dashboard.html', {
    'total': total,
    'avg_cgpa': round(avg_cg, 2),
  })

urlpatterns = [
  path('', StudentListView.as_view(),
    name='student-list'),
  path('update/<int:pk>/', update_student),
  path('delete/<int:pk>/', delete_student),
  path('dashboard/', dashboard),
]
